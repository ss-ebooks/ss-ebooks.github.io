วาดวงรีด้วย Midpoint ellipse algorithm

วิธีการนี้แบ่งเส้นรอบรูปออกเป็น 4 quadrants การคำนวณหาตำแหน่งของ pixel จะทำเฉพาะแต่ใน quadrant ที่ 1 เท่านั้น แล้วโยงไป ยัง quadrant อื่นด้วยความสัมพันธ์ตามที่แสดงในรูป

การคำนวณจะแบ่งออกเป็น 2 ตอน ตอนแรกเป็นคำนวณหาตำแหน่งของ pixel ใน region 1 ก่อน แล้วค่อยหาตำแหน่งใน region 2 ในตอนที่ 2

decision variables ที่ใช้เลือกตำแหน่งของ pixel มีดังนี้

  • d,dx,dy d คือ decision variables สำหรับ dx,dy แยกออกมาเพื่อความสะดวกในการคำนวณ
  • rx​ คือ radius บน X-axis
  • ry​ คือ radius บน Y-axis
  • rx​,ry​ อาจพิจารณาว่าเป็น semi-major-axis หรือ semi-minor-axis ก็ได้ขึ้นกับค่าที่มอบให้
    • rx​>ry​ จะได้วงรีแนวนอน
    • rx​<ry​ จะได้วงรีแนวตั้ง
    • rx​=ry​ จะได้วงกลม

Algorithm สำหรับ region 1

  1. รับค่า cx,cy,rx,ry

  2. กำหนดค่าเริ่มต้น x=0,y=ry

  3. กำหนดค่าเริ่มต้นสำหรับ decision variables dk​dkx​dky​​=ry2​+41​rx2​−rx2​ry​=2ry2​x=2rx2​y​

  4. plot pixel ตำแหน่ง (x,y)

  5. การปรับแต่งค่า decision variable และ ตำแหน่งของ pixel ต่อไป

  • dk​<0 : xk+1​yk+1​dk+1x​dk+1​​=xk​+1=yk​=dkx​+2ry2​=dk​+dk+1x​+ry2​​

  • dk​<=0 : xk+1​yk+1​dk+1x​dk+1y​dk+1​​=xk​+1=yk​−1=dkx​+2ry2​=dky​−2rx2​=dk​+dk+1x​+ry2​​

  1. ทำซ้ำขั้นตอน 4 - 5 จนกว่า dkx​≥dky​

Algorithm สำหรับ region 2

ใช้ตำแหน่ง (x,y) สุดท้ายจาก region 1 เป็นต่าเริ่มต้น

1.กำหนดค่าเริ่มต้นสำหรับ decision variables dk​=ry2​(x+0.5)2+rx2​(y−1)2−rx2​ry2​

2.plot pixel ตำแหน่ง (x,y)

3.การปรับแต่งค่า decision variable และ ตำแหน่งของ pixel ต่อไป

  • dk​>0 : yk+1​xk+1​dk+1y​dk+1​​=yk​−1=xk​=dky​−2rx2​=rx2​−dk+1y​​

  • dk​<=0 : xk+1​yk+1​dk+1x​dk+1y​dk+1​​=xk​+1=yk​−1=dkx​+2ry2​=dky​−2rx2​=dk​+dk+1x​−dk+1y​+rx2​​

  1. ทำซ้ำขั้นตอน 2 - 3 จนกว่า y<0

    void plot(int x,int y,Color clr){
	DrawPixel(cx+x,cy+y,clr);
	DrawPixel(cx-x,cy+y,clr);
	DrawPixel(cx-x,cy-y,clr);
	DrawPixel(cx+x,cy-y,clr);
}

void draw_ellipse(){
	/* TODO : draw a frame */
	int x = 0;
	int y = ry;

	// region 1
	int d1 = pow(ry,2) + 0.25*pow(rx,2) - ry*pow(rx,2);
	int dx = 2*x*pow(ry,2);
	int dy = 2*y*pow(rx,2);

	while(dx < dy){
		plot(x,y,RED);
		x++;
		if(d1<0){
			dx += (2*pow(ry,2));
			d1 += dx + pow(ry,2);
		}else{
			y--;
			dx += (2*pow(ry,2));
			dy -= (2*pow(rx,2));
			d1 += dx - dy +pow(ry,2);
		}
	}

	//region 2
	// use last (x,y) of region 1 as initial value

	int d2 = pow(ry,2) * pow(x+0.5,2) + pow(rx,2)*pow(y-1,2) - pow(rx,2)*pow(ry,2);

	while(y>=0){
		plot(x,y,RED);
		y--;
		if(d2>0){
			dy -= 2*pow(rx,2);
			d2 += pow(rx,2) - dy;
		}else{
			x++;
			dx += 2*pow(ry,2);
			dy -= 2*pow(rx,2);
			d2 += dx - dy + pow(rx,2);
		}
	}
}
