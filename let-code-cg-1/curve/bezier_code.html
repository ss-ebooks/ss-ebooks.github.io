Let's Code Graphics on Raspberry Pi

วาดเส้นโค้ง Bezier curve ด้วย raylib

โปรแกรมนี้จะใช้สมการ 1.12 เป็น algorithm ในการวาด Quadratic Bezier Curve

ติดตั้ง Linear algebra library

เนื่องจากต้องใช้การทำงานกับ matrix ดังนั้นจำเป็นมีการใช้งาน linear algebra library ซึ่งมีให้เลือกมากมาย ในโครงการนี้และโครงการอื่นต่อจากนี้ เลือกใช้ Armadillo

ก่อนทำการติดตั้ง armadillo ต้องติดตั้ง required libraries ก่อนดังนี้

sudo apt install libopenblas-dev, liblapack-dev, libarpack2-dev, libsuperlu-dev

การติดตั้ง armadillo สามารถทำตามขั้นตอนที่กล่าวไว้ในเอกสารการติดตั้ง ซึ่งจะเป็นรุ่นล่าสุด หรือใช้การติดตั้งจาก repository (อาจไม่ใช่รุ่นล่าสุด) ด้วยคำสั่ง 

sudo apt install libarmadillo-dev
หมายเหตุ

raylib เองก็มี raymath เป็น library ทางด้าน linear algebra อยู่ แต่มีข้อจำกัดอยู่ ไม่สะดวกเหมือนกับการใช้ library ที่ออกแบบมาโดยตรง

การสร้างชุดคำสั่ง

  1. กำหนด header files โดยมีการเพิ่ม header file ของ armadillo เข้ามา
#include "raylib.h"
#include "armadillo"
#include "vector"
  1. ประกาศตัวแปร
arma::mat M(4,4);
arma::mat P(2,4);
arma::mat PM(2,4);
std::vector<Vector2> drawPoints;
  • M เป็นตัวแปรประเภท matrix มีขนาด 4×4 (row X column) ใช้เป็น coefficient matrix
  • P เป็นตัวแปรประเภท matrix มีขนาด 2×4 ใช้เป็น control points matrix
  • PM เป็นตัวแปรประเภท matrix มีขนาด 2×4 ใช้เป็นที่พักของผลลัพธ์ P⋅M
  • drawPoints เป็นตัวแปรประเภท vector ใช้เก็บ coordinates ของผลลัพธ์จาก interpolation
  1. การทำ Interpolation ชุดคำสั่งอยู่ใน setup() 
void setup(){
	/* TODO : initialize all variables */
	
	/* interpolat points to draw */
	M  = arma::mat("1 -3 3 -1;0 3 -6 3 ;0 0 3 -3;0 0 0 1");
	P  = arma::mat("200 300 400 500;100 300 100 400");
	PM = P*M;
	for(double t=0;t<=1;t+=0.01){
		arma::mat T(4,1);
		arma::mat Vt(2,1);
		T(3,0) = t*t*t;
		T(2,0) = t*t;
		T(1,0) = t;
		T(0,0) = 1; 
		Vt = PM * T;
		drawPoints.push_back((Vector2){(float)Vt(0,0),(float)Vt(1,0)});
	}
}
  1. วาดเส้นโค้ง และ control points
void draw(){
	/* TODO : draw a frame */
	/* draw curve */
	Vector2 *data = drawPoints.data();
	DrawLineStrip(data,drawPoints.size(),RED);
	
	/* draw control points */
	DrawCircleV((Vector2){(float)P(0,0),(float)P(1,0)},10.0f,BLUE);	
	DrawCircleV((Vector2){(float)P(0,1),(float)P(1,1)},10.0f,BLUE);	
	DrawCircleV((Vector2){(float)P(0,2),(float)P(1,2)},10.0f,BLUE);	
	DrawCircleV((Vector2){(float)P(0,3),(float)P(1,3)},1 0.0f,BLUE);
}

ชุดฟังก์ชั่นของ raylib

raylib มีเพียงฟังก์ชันสำหรับการวาด Bezier curve คือ 

DrawLineBezier(Vector2 startPos, Vector2 endPos, float thick, Color color);

ดูจาก arguments จะไม่มีการกำหนดตำแหน่ง control points นั่นคือทำให้รูปร่างของเส้นโค้งที่เกิดจากฟังก์ชั่นนี้มีลักษณะเป็น S-shape อย่างเดียว
